### Начало проэкта
mkdir <folder_project_name> && cd <folder_project_name>
pip install virtualenv
virtualenv venv
mkdir app && cd app
git init
git status
pip install flask
python -c "import flask; print(flask.__version__)"

# Разработка приложения
vi main.py
vi wsgi.py
mkdir templates
vi templates/index.html
vi templates/base.html

# Запуск приложения
export FLASK_APP=app
export FLASK_ENV=development
flask run

# Создать репозиторий на GitHub
git remote add origin git@github.com:makotakara/<folder_project_name>.git
pip freeze > requirement.txt
vi .gitignore
git branch -M main
git add .
git commit -am "firts commit"
git push -u origin main

# Для разворачивания проэкта на другой сервере
# Зайти на серевер 
cd /home/app_user/project/

virtualenv venv
source venv/bin/activate
git clone https://github.com/makotakara/performance_page.git
mv performance_page app && cd app

# Скачать новую версию проэкта
